**2. `fit_with_retry(model_fn, model_kwargs, *, seed, num_warmup_first=2000, num_warmup_retry=4000, num_samples=2000, num_chains=4, target_accept_first=0.95, target_accept_retry=0.99, max_tree_depth=10) -> tuple[MCMC, str, list[dict]]`:**

Implements the retry loop from RESEARCH.md Section 5. `model_kwargs` is a dict of keyword arguments passed to `mcmc.run(rng_key, **model_kwargs, extra_fields=('diverging',))`.

```python
attempts: list[dict] = []
for attempt_idx, (n_warmup, target_accept) in enumerate([
    (num_warmup_first, target_accept_first),
    (num_warmup_retry, target_accept_retry),
]):
    kernel = NUTS(model_fn, target_accept_prob=target_accept, max_tree_depth=max_tree_depth)
    mcmc = MCMC(kernel, num_warmup=n_warmup, num_samples=num_samples,
                num_chains=num_chains, progress_bar=False)
    mcmc.run(jax.random.PRNGKey(seed + attempt_idx), **model_kwargs,
             extra_fields=('diverging',))
    diag = run_diagnostics(mcmc)
    record = {'attempt': attempt_idx + 1, 'num_warmup': n_warmup,
              'target_accept_prob': target_accept, **diag}
    attempts.append(record)
    if diag['passes_gate']:
        return mcmc, 'PASS', attempts
return mcmc, 'FAILED', attempts
```

**3. `make_fit_summary(mcmc, *, status, attempts, var_names, **meta) -> dict`:**

Trimmed per-fit JSON shape (~10 KB). `meta` is arbitrary metadata (e.g. `subject_id`, `condition`, `seed`, `rnn_seed_idx`).

**M3 fix (ArviZ kwarg):** ArviZ 0.23.4's `az.summary` does NOT accept `kind=`. Use `stat_focus='stats'` / `stat_focus='diagnostics'` (per RESEARCH.md Section 5 canonical pattern) — using `kind=` raises TypeError at runtime. Implementation:

```python
idata = az.from_numpyro(mcmc)
# M3: use stat_focus, not kind. Two calls — one for stats, one for diagnostics.
summary_stats = az.summary(idata, var_names=var_names, stat_focus='stats', round_to='none')
summary_diag = az.summary(idata, var_names=var_names, stat_focus='diagnostics', round_to='none')
param_stats = {}
for param in var_names:
    row = summary_stats.loc[param]
    hdi = az.hdi(idata, hdi_prob=0.95, var_names=[param])
    posterior_arr = np.asarray(idata.posterior[param])
    param_stats[param] = {
        'mean': float(row['mean']),
        'median': float(np.median(posterior_arr)),
        'sd': float(row['sd']),
        'hdi_2.5': float(hdi[param].values[0]),
        'hdi_97.5': float(hdi[param].values[1]),
        'rhat': float(summary_diag.loc[param]['r_hat']),
        'ess_bulk': float(summary_diag.loc[param]['ess_bulk']),
    }

div_field = mcmc.get_extra_fields().get('diverging')
n_divergences = int(div_field.sum()) if div_field is not None else None

return {
    **meta,
    'status': status,           # 'PASS' or 'FAILED'
    'attempts': attempts,
    'params': param_stats,
    'n_divergences': n_divergences,
    'n_chains': mcmc.num_chains,
    'n_samples_per_chain': mcmc.num_samples,
}
```
